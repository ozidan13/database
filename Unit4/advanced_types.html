🗂️ JSONB — البيانات شبه المنظّمة

JSONB — Binary JSON with Full Indexing and Query Power

الـ JSON والـ JSONB كلاهما يخزّن JSON. الفرق: JSONB بيخزّنه في صيغة Binary محسّنة — أسرع في القراءة، يدعم الـ Indexing الكامل، ويتجاهل المسافات الزائدة والتكرار في المفاتيح. استخدم JSONB دايماً إلا لو محتاج للحفاظ على ترتيب المفاتيح أو سرعة الكتابة بالضبط.

🛒 متى تستخدم JSONB؟

✅ خصائص المنتجات التي تختلف بين الفئات (موبايل: RAM/Storage، ملابس: Size/Color)
✅ إعدادات المستخدم المتنوعة (preferences, notifications)
✅ بيانات API خارجية بتيجي محتلفة كل مرة
✅ Metadata متغيّر لا تعرف أعمدته مسبقاً

❌ مش مناسب لو البيانات منظّمة ثابتة — استخدم أعمدة عادية

JSONB Operators

->
جيب قيمة مفتاح → يرجع JSONB
data->'name' → "أحمد"
->>
جيب قيمة مفتاح → يرجع TEXT
data->>'name' → أحمد
#>
مسار متداخل → JSONB
data#>'{address,city}'
#>>
مسار متداخل → TEXT
data#>>'{address,city}'
@>
يحتوي على؟ (Contains)
data @> '{"color":"red"}'
<@
محتوى في؟ (Contained by)
'{"a":1}' <@ data
?
المفتاح موجود؟
data ? 'email'
?|
أي من المفاتيح موجود؟
data ?| array['a','b']
JSONB — Complete Reference
-- ══ إنشاء جدول مع JSONB ══ CREATE TABLE product ( product_id BIGSERIAL PRIMARY KEY, name TEXT NOT NULL, price NUMERIC(10,2), attributes JSONB DEFAULT '{}' -- خصائص مرنة ); -- ══ INSERT بيانات JSONB ══ INSERT INTO product (name, price, attributes) VALUES ('iPhone 15', 4999, '{"brand":"Apple","ram":6,"storage":[128,256,512],"colors":["black","white"]}'), ('T-Shirt', 199, '{"sizes":["S","M","L","XL"],"material":"cotton","colors":["red","blue"]}'); -- ══ الاستعلام من JSONB ══ SELECT name, attributes->'brand' AS brand_jsonb, -- يرجع "Apple" (مع quotes) attributes->>'brand' AS brand_text, -- يرجع Apple (بدون quotes) attributes->>'ram' AS ram, attributes->'colors'->>0 AS first_color -- أول لون في array FROM product WHERE attributes->>'brand' = 'Apple'; -- ══ WHERE مع JSONB Operators ══ -- المنتجات اللي اللون الأحمر متاح فيها SELECT name FROM product WHERE attributes @> '{"colors":["red"]}'; -- المنتجات اللي عندها خاصية ram SELECT name FROM product WHERE attributes ? 'ram'; -- ══ JSONB UPDATE — jsonb_set ══ UPDATE product SET attributes = jsonb_set(attributes, '{ram}', '8') WHERE product_id = 1; -- إضافة مفتاح جديد UPDATE product SET attributes = attributes || '{"warranty_years":2}'::jsonb WHERE product_id = 1; -- حذف مفتاح UPDATE product SET attributes = attributes - 'ram' -- يحذف مفتاح ram WHERE product_id = 1; -- ══ jsonb_agg — دمج JSONB من عدة صفوف ══ SELECT category_id, jsonb_agg(jsonb_build_object( 'id', product_id, 'name', name, 'price', price )) AS products_json FROM product GROUP BY category_id; -- ══ GIN Index للـ JSONB ══ CREATE INDEX idx_product_attrs ON product USING GIN (attributes); -- يُسرّع @>, ?, ?|, ?& operators -- لو عايز index على مفتاح معين: CREATE INDEX idx_product_brand ON product ((attributes->>'brand'));

PostgreSQL Arrays — Multidimensional Array Support

Arrays — Full Reference
-- ══ تعريف Arrays ══ CREATE TABLE post ( post_id BIGSERIAL PRIMARY KEY, title TEXT, tags TEXT[] DEFAULT '{}', -- Array of TEXT scores INT[] -- Array of INT ); -- ══ INSERT Arrays ══ INSERT INTO post (title, tags, scores) VALUES ('درس SQL', '{database,sql,postgres}', '{95,88,91}'), ('درس Python', ARRAY['python','programming']', '{80,92}'); -- ══ الوصول للعناصر (1-indexed في PostgreSQL!) ══ SELECT title, tags[1] AS first_tag, tags['2'::INT] AS second_tag, tags[1:2] AS first_two_tags -- slice FROM post; -- ══ الاستعلام على Arrays ══ -- المقالات اللي فيها تاج 'sql' SELECT title FROM post WHERE 'sql' = ANY(tags); -- هل 'sql' موجود في array؟ -- المقالات اللي كل تاجاتها موجودة في قائمة معينة SELECT title FROM post WHERE tags @> ARRAY['sql','database']; -- Contains: مقالة فيها sql وdatabase -- التقاطع: مقالات فيها أي من هذه التاجات SELECT title FROM post WHERE tags && ARRAY['sql','python']; -- && = Overlap operator -- ══ Array Functions ══ SELECT array_length(tags, 1) AS tag_count, array_append(tags, 'new_tag') AS with_new, array_remove(tags, 'sql') AS without_sql, array_to_string(tags, ', ') AS tags_str, array_cat(tags, ARRAY['extra']) AS combined, UNNEST(tags) AS expanded_tag -- يحوّل array لصفوف FROM post; -- ══ UNNEST — تحويل Array لجدول ══ SELECT post_id, title, UNNEST(tags) AS tag FROM post ORDER BY post_id, tag; -- ══ GIN Index للـ Arrays === CREATE INDEX idx_post_tags ON post USING GIN (tags); -- يُسرّع @>, &&, <@

ENUM — Type-Safe Categorical Values

ENUM — Definition, Use, and Modifications
-- ══ إنشاء ENUM Type ══ CREATE TYPE order_status_enum AS ENUM ( 'pending', 'confirmed', 'shipped', 'delivered', 'cancelled', 'refunded' ); CREATE TYPE employee_level_enum AS ENUM ( 'junior', 'mid', 'senior', 'lead', 'principal' ); -- ══ استخدام ENUM في جدول ══ CREATE TABLE orders ( order_id BIGSERIAL PRIMARY KEY, status order_status_enum NOT NULL DEFAULT 'pending' ); -- ══ فوائد ENUM ══ -- 1. Type safety: ما تقدرش تدخّل قيمة خارج القائمة UPDATE orders SET status = 'lost'; -- ERROR: invalid input value -- 2. الترتيب الطبيعي: VALUES مرتّبة حسب تعريفها SELECT * FROM orders WHERE status > 'pending'::order_status_enum; -- بيرجع confirmed, shipped, delivered... -- 3. الـ Storage: أصغر من TEXT (يخزّن 4 bytes كل قيمة) -- ══ تعديل ENUM ══ -- إضافة قيمة جديدة (بدون rebuild الجدول) ALTER TYPE order_status_enum ADD VALUE 'processing' AFTER 'confirmed'; -- ⚠️ مش ممكن تحذف قيمة من ENUM في PostgreSQL! -- الحل: إنشاء type جديد + تحويل البيانات + حذف القديم CREATE TYPE new_status AS ENUM('pending', 'delivered'); -- بدون المحذوفة ALTER TABLE orders ALTER COLUMN status TYPE new_status USING status::TEXT::new_status; -- ══ قائمة ENUMs الموجودة ══ SELECT typname, array_agg(enumlabel ORDER BY enumsortorder) AS values FROM pg_type JOIN pg_enum USING (oid) GROUP BY typname;

UUID — Generation and Usage
-- ══ UUID Extension ══ CREATE EXTENSION IF NOT EXISTS "pgcrypto"; -- أو في PG 14+: CREATE EXTENSION IF NOT EXISTS "uuid-ossp"; -- ══ UUID كـ Primary Key ══ CREATE TABLE api_token ( token_id UUID DEFAULT gen_random_uuid() PRIMARY KEY, user_id BIGINT NOT NULL REFERENCES users, token_name TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), expires_at TIMESTAMPTZ ); -- توليد UUID يدوياً SELECT gen_random_uuid(); -- → a4d1f8b2-7c3e-4f2a-8b9c-1d5e7f3a9b2c -- ══ UUID vs BIGSERIAL ══ -- UUID: -- ✅ فريد عالمياً — آمن في الـ API (مش ممكن يتخمّن) -- ✅ ممكن تولّده في الـ Client قبل الحفظ -- ✅ لا مشاكل في الـ Merge / Distributed systems -- ❌ أكبر حجماً (16 bytes vs 8) -- ❌ بطيء في الـ Index (random inserts) -- BIGSERIAL: -- ✅ أسرع في الـ Index (sequential) -- ✅ أصغر حجماً -- ❌ يمكن تخمينه — لا تعرضه في الـ API -- ❌ مشاكل في الـ Distributed systems
🔍 Full-Text Search — البحث النصي الكامل

tsvector, tsquery, GIN Index — Built-in Search Engine

PostgreSQL بيدعم بحثاً نصياً متكاملاً — tokenization، stemming، ranking — بدون الحاجة لـ Elasticsearch في كتير من الحالات.

Full-Text Search — Complete Setup
-- ══ المفاهيم ══ -- tsvector: نص محوَّل لـ tokens قابلة للبحث -- tsquery: شرط البحث -- @@ : مطابقة tsvector مع tsquery -- ══ بحث بسيط ══ SELECT to_tsvector('english', 'The quick brown fox'); -- → 'brown':3 'fox':4 'quick':2 (the مش مهم — stop word) SELECT to_tsquery('english', 'fox & quick'); -- → 'fox' & 'quick' -- ══ بحث في جدول ══ SELECT title, body FROM article WHERE to_tsvector('arabic', title || ' ' || body) @@ plainto_tsquery('arabic', 'قاعدة بيانات'); -- ══ Ranking النتائج ══ SELECT title, ts_rank( to_tsvector('english', body), to_tsquery('english', 'database') ) AS relevance FROM article WHERE to_tsvector('english', body) @@ to_tsquery('english', 'database') ORDER BY relevance DESC LIMIT 10; -- ══ Setup الأمثل: عمود tsvector محفوظ ══ ALTER TABLE article ADD COLUMN search_vector TSVECTOR; -- تحديث بالبيانات الموجودة UPDATE article SET search_vector = to_tsvector('english', COALESCE(title,'') || ' ' || COALESCE(body,'')); -- GIN Index على العمود المحفوظ (أسرع بكتير) CREATE INDEX idx_article_fts ON article USING GIN (search_vector); -- Trigger لتحديث تلقائي عند INSERT/UPDATE CREATE OR REPLACE FUNCTION update_article_fts() RETURNS TRIGGER AS $$ BEGIN NEW.search_vector := to_tsvector('english', COALESCE(NEW.title, '') || ' ' || COALESCE(NEW.body, '')); RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER trig_article_fts BEFORE INSERT OR UPDATE ON article FOR EACH ROW EXECUTE FUNCTION update_article_fts();

🛒 تمرين: منتجات متجر بخصائص مرنة

السيناريو

قاعدة بيانات متجر فيها منتجات ذات خصائص مختلفة (إلكترونيات لها RAM/Storage، ملابس لها مقاسات).

  • Aأنشئ جدول product مع ENUM للحالة، UUID كـ primary key، وJSONB للخصائص.
  • Bأضف GIN index على attributes، واجيب كل المنتجات اللي فيها لون "أبيض".
  • Cاستخدم UNNEST لاستخراج كل مقاس من حقل sizes (Array) مع اسم المنتج.
Advanced Types — Solutions
-- A: جدول product CREATE TYPE product_status AS ENUM ('active', 'inactive', 'discontinued'); CREATE TABLE product ( product_id UUID DEFAULT gen_random_uuid() PRIMARY KEY, name TEXT NOT NULL, price NUMERIC(10,2) NOT NULL, status product_status DEFAULT 'active', attributes JSONB DEFAULT '{}' ); -- B: GIN Index + استعلام CREATE INDEX idx_product_attrs ON product USING GIN (attributes); SELECT name, attributes->>'brand' AS brand FROM product WHERE attributes @> '{"colors":["أبيض"]}'; -- C: UNNEST للمقاسات SELECT name, UNNEST((attributes->'sizes')::TEXT[]) AS size FROM product WHERE attributes ? 'sizes' ORDER BY name, size;

١. الفرق الرئيسي بين JSON وJSONB في PostgreSQL؟
JSONB أحدث وبس
JSONB يخزّن بصيغة Binary محسّنة — يدعم GIN Index، أسرع قراءة، يتجاهل تكرار المفاتيح. JSON نص خام — أسرع كتابة، يحتفظ بالترتيب.
مفيش فرق في الاستخدام
٢. ما الـ Operator المناسب للبحث "المنتجات اللي فيها اللون الأحمر ضمن attributes"؟
attributes->>'color' = 'red'
attributes @> '{"colors":["red"]}' — الـ @> يتحقق من الاحتواء في JSONB
attributes ? 'red'
٣. ما ميزة ENUM على TEXT CHECK CONSTRAINT؟
ENUM أسرع في الكتابة
ENUM يوفّر Storage أصغر، ترتيب طبيعي للقيم، وقابلية للتوسع بدون Rebuild — وهو Type مستقل قابل للمشاركة بين الجداول.
TEXT CHECK CONSTRAINT أفضل دايماً

UNNEST() — بتحوّل كل عنصر في الـ Array لصف منفصل
