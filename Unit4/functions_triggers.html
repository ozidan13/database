⚙️ PL/pgSQL — لغة البرمجة داخل PostgreSQL

PL/pgSQL — Procedural Language for Server-Side Logic

PL/pgSQL هي لغة إجرائية كاملة داخل PostgreSQL. بتكتب functions مع variables، conditions، loops، وexception handling — وتتنفّذ داخل الـ Database Engine مباشرة.

🏢 متى تكتب Function بدل SQL عادي؟

✅ منطق معقد يحتاج شروط ومتغيرات
✅ عمليات متكررة بتتعمل من أكتر من مكان
✅ بدك تتجنب نقل بيانات ضخمة للـ Application
✅ Triggers — لا يمكن إنشاؤهم بدون Function
✅ Business rules لازم تُطبَّق على مستوى الـ DB

❌ لا تضع كل منطق التطبيق في DB — يصعّب الصيانة

PL/pgSQL — Function Template
-- ══ هيكل Function الأساسي ══ CREATE OR REPLACE FUNCTION function_name( param1 DATA_TYPE, -- Input parameter param2 DATA_TYPE DEFAULT value -- Optional with default ) RETURNS RETURN_TYPE -- VOID, INT, TEXT, TABLE(...)... LANGUAGE plpgsql SECURITY DEFINER -- اختياري: تشتغل بصلاحيات المنشئ STABLE -- أو VOLATILE (default) أو IMMUTABLE AS $$ DECLARE var1 INT; -- تعريف المتغيرات var2 TEXT := 'hello'; -- مع قيمة أولية rec RECORD; -- لأي صف/نوع BEGIN -- جسم الدالة RETURN result; -- القيمة المعادة END; $$ ; -- ══ Function بسيطة ══ CREATE OR REPLACE FUNCTION calculate_age(birth_date DATE) RETURNS INT LANGUAGE plpgsql IMMUTABLE -- نفس المدخلات = نفس النتيجة دايماً AS $$ BEGIN RETURN EXTRACT(YEAR FROM AGE(CURRENT_DATE, birth_date))::INT; END; $$; -- استخدامها مثل أي function عادية SELECT full_name, calculate_age(birth_date) AS age FROM employee; -- ══ Function ترجع جدول (RETURNS TABLE) ══ CREATE OR REPLACE FUNCTION get_dept_employees(p_dept_id INT) RETURNS TABLE( emp_id INT, name TEXT, salary NUMERIC ) LANGUAGE plpgsql AS $$ BEGIN RETURN QUERY SELECT e.emp_id, e.full_name, e.salary FROM employee e WHERE e.dept_id = p_dept_id AND e.is_active = TRUE ORDER BY e.salary DESC; END; $$; -- الاستدعاء SELECT * FROM get_dept_employees(3);

Exception Handling — Production Pattern
-- ══ هيكل Exception ══ BEGIN -- الكود الأساسي INSERT INTO orders (...) VALUES (...); EXCEPTION WHEN unique_violation THEN RAISE NOTICE 'Duplicate key: %', SQLERRM; WHEN foreign_key_violation THEN RAISE EXCEPTION 'Invalid FK: %', SQLERRM; WHEN division_by_zero THEN RETURN NULL; WHEN others THEN -- catch all RAISE EXCEPTION 'Error: % (Code: %)', SQLERRM, SQLSTATE; END; -- ══ Function مع Exception للـ UPSERT المخصص ══ CREATE OR REPLACE FUNCTION create_or_update_customer( p_email TEXT, p_name TEXT, p_phone TEXT ) RETURNS BIGINT LANGUAGE plpgsql AS $$ DECLARE v_id BIGINT; BEGIN INSERT INTO customer (email, full_name, phone) VALUES (p_email, p_name, p_phone) ON CONFLICT (email) DO UPDATE SET full_name = EXCLUDED.full_name, phone = EXCLUDED.phone, updated_at = NOW() RETURNING customer_id INTO v_id; RETURN v_id; EXCEPTION WHEN check_violation THEN RAISE EXCEPTION 'Validation failed for email: %', p_email; WHEN not_null_violation THEN RAISE EXCEPTION 'Required fields missing'; END; $$; -- ══ Raise مستويات مختلفة ══ RAISE DEBUG 'Debug info: %', val; -- للـ Development RAISE NOTICE 'Notice: %', val; -- للمعلومات RAISE WARNING 'Warning: %', val; -- تحذير RAISE EXCEPTION 'Error: %', val; -- يوقف التنفيذ
⚡ Triggers — المحفّزات التلقائية

Triggers — Automatic Logic on INSERT/UPDATE/DELETE

الـ Trigger هو function بتُنفَّذ تلقائياً عند INSERT أو UPDATE أو DELETE على جدول معين. لا تحتاج التطبيق يستدعيها.

BEFORE

بيُنفَّذ قبل العملية. ممكن تعدّل NEW أو تلغي العملية بـ RETURN NULL.

AFTER

بيُنفَّذ بعد العملية. مناسب للـ Audit Logs وتحديث جداول مرتبطة.

INSTEAD OF

بدّل العملية بالكامل. للـ Views غير القابلة للتعديل فقط.

Triggers — Complete Patterns
-- ══ Pattern 1: Auto-update updated_at ══ CREATE OR REPLACE FUNCTION fn_set_updated_at() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at := NOW(); RETURN NEW; -- BEFORE trigger: يجب RETURN NEW END; $$ LANGUAGE plpgsql; CREATE TRIGGER trig_customer_updated_at BEFORE UPDATE ON customer FOR EACH ROW EXECUTE FUNCTION fn_set_updated_at(); -- ══ Pattern 2: Validation Trigger ══ CREATE OR REPLACE FUNCTION fn_validate_salary() RETURNS TRIGGER AS $$ BEGIN IF NEW.salary < 0 THEN RAISE EXCEPTION 'Salary cannot be negative: %', NEW.salary; END IF; IF NEW.salary > 500000 THEN RAISE EXCEPTION 'Salary exceeds maximum allowed'; END IF; RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER trig_validate_salary BEFORE INSERT OR UPDATE ON employee FOR EACH ROW WHEN (NEW.salary IS NOT NULL) -- Conditional trigger EXECUTE FUNCTION fn_validate_salary(); -- ══ Pattern 3: Soft Delete Trigger ══ CREATE OR REPLACE FUNCTION fn_soft_delete() RETURNS TRIGGER AS $$ BEGIN UPDATE customer SET deleted_at = NOW(), is_active = FALSE WHERE customer_id = OLD.customer_id; RETURN NULL; -- يمنع الحذف الفعلي END; $$ LANGUAGE plpgsql; CREATE TRIGGER trig_soft_delete_customer BEFORE DELETE ON customer FOR EACH ROW EXECUTE FUNCTION fn_soft_delete(); -- ══ OLD و NEW في الـ Triggers ══ -- INSERT: NEW موجود, OLD مش موجود -- UPDATE: OLD (قبل) و NEW (بعد) موجودين -- DELETE: OLD موجود, NEW مش موجود -- ══ إدارة الـ Triggers ══ ALTER TABLE customer DISABLE TRIGGER trig_customer_updated_at; ALTER TABLE customer ENABLE TRIGGER trig_customer_updated_at; DROP TRIGGER IF EXISTS trig_customer_updated_at ON customer;
📋 Audit Log Pattern — سجل التغييرات

Complete Audit Trail — Track Every Change with Triggers

Audit Log — Production Pattern
-- ══ جدول الـ Audit Log ══ CREATE TABLE audit_log ( log_id BIGSERIAL PRIMARY KEY, table_name TEXT NOT NULL, operation TEXT NOT NULL, -- INSERT/UPDATE/DELETE record_id BIGINT, old_data JSONB, -- الصف قبل التغيير new_data JSONB, -- الصف بعد التغيير changed_by TEXT DEFAULT CURRENT_USER, changed_at TIMESTAMPTZ DEFAULT NOW() ); -- ══ Generic Audit Function ══ CREATE OR REPLACE FUNCTION fn_audit_log() RETURNS TRIGGER AS $$ BEGIN IF (TG_OP = 'INSERT') THEN INSERT INTO audit_log (table_name, operation, record_id, new_data) VALUES (TG_TABLE_NAME, TG_OP, NEW.id, to_jsonb(NEW)); RETURN NEW; ELSIF (TG_OP = 'UPDATE') THEN INSERT INTO audit_log (table_name, operation, record_id, old_data, new_data) VALUES (TG_TABLE_NAME, TG_OP, NEW.id, to_jsonb(OLD), to_jsonb(NEW)); RETURN NEW; ELSIF (TG_OP = 'DELETE') THEN INSERT INTO audit_log (table_name, operation, record_id, old_data) VALUES (TG_TABLE_NAME, TG_OP, OLD.id, to_jsonb(OLD)); RETURN OLD; END IF; END; $$ LANGUAGE plpgsql; -- تطبيقه على كل جدول محتاجه CREATE TRIGGER trig_audit_employee AFTER INSERT OR UPDATE OR DELETE ON employee FOR EACH ROW EXECUTE FUNCTION fn_audit_log(); CREATE TRIGGER trig_audit_orders AFTER INSERT OR UPDATE OR DELETE ON orders FOR EACH ROW EXECUTE FUNCTION fn_audit_log(); -- ══ استعلام من الـ Audit Log ══ -- كل التغييرات على موظف معين SELECT TO_CHAR(changed_at, 'YYYY-MM-DD HH24:MI') AS when, operation, changed_by, old_data->>'salary' AS old_salary, new_data->>'salary' AS new_salary FROM audit_log WHERE table_name = 'employee' AND record_id = 42 ORDER BY changed_at DESC;

Banking System — Solutions
-- A: Transfer Function مع Exception Handling CREATE OR REPLACE FUNCTION transfer_funds( p_from BIGINT, p_to BIGINT, p_amount NUMERIC ) RETURNS TEXT LANGUAGE plpgsql AS $$ DECLARE v_from_balance NUMERIC; v_is_frozen BOOLEAN; BEGIN IF p_amount <= 0 THEN RAISE EXCEPTION 'Amount must be positive'; END IF; SELECT balance, is_frozen INTO v_from_balance, v_is_frozen FROM account WHERE account_id = p_from FOR UPDATE; -- Lock the row IF NOT FOUND THEN RAISE EXCEPTION 'Source account not found'; END IF; IF v_is_frozen THEN RAISE EXCEPTION 'Account is frozen'; END IF; IF v_from_balance < p_amount THEN RAISE EXCEPTION 'Insufficient balance. Available: %', v_from_balance; END IF; UPDATE account SET balance = balance - p_amount WHERE account_id = p_from; UPDATE account SET balance = balance + p_amount WHERE account_id = p_to; INSERT INTO transaction (from_account, to_account, amount, status) VALUES (p_from, p_to, p_amount, 'completed'); RETURN 'Transfer successful'; EXCEPTION WHEN others THEN INSERT INTO transaction (from_account, to_account, amount, status) VALUES (p_from, p_to, p_amount, 'failed'); RAISE; END; $$; -- B: Audit Trigger على balance CREATE OR REPLACE FUNCTION fn_audit_balance() RETURNS TRIGGER AS $$ BEGIN IF OLD.balance != NEW.balance THEN INSERT INTO audit_log(table_name, operation, record_id, old_data, new_data) VALUES('account', 'BALANCE_CHANGE', NEW.account_id, jsonb_build_object('balance', OLD.balance), jsonb_build_object('balance', NEW.balance)); END IF; RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER trig_audit_account_balance AFTER UPDATE ON account FOR EACH ROW EXECUTE FUNCTION fn_audit_balance(); -- C: Customer Portfolio Summary CREATE OR REPLACE FUNCTION get_customer_portfolio(p_owner_id BIGINT) RETURNS TABLE(account_type TEXT, total_balance NUMERIC, account_count INT) LANGUAGE plpgsql AS $$ BEGIN RETURN QUERY SELECT a.account_type, SUM(a.balance), COUNT(*)::INT FROM account a WHERE a.owner_id = p_owner_id AND a.is_frozen = FALSE GROUP BY a.account_type; END; $$;
🧠 اختبار الفهم
١. في Trigger من نوع BEFORE UPDATE — لو رجعنا NULL من الـ Function — ماذا يحدث؟
يُنفَّذ الـ UPDATE بشكل طبيعي
يُلغى الـ UPDATE تماماً — RETURN NULL في BEFORE trigger = يوقف العملية
يحدث Error
٢. ما الفرق بين FOR EACH ROW وFOR EACH STATEMENT في Trigger؟
مفيش فرق
FOR EACH ROW يُنفَّذ مرة لكل صف. FOR EACH STATEMENT يُنفَّذ مرة واحدة فقط للـ SQL Statement كله — حتى لو عدّل 1000 صف.
FOR EACH STATEMENT أسرع دايماً
٣. في Trigger من نوع DELETE — ما المتغير المتاح: NEW أم OLD؟
NEW فقط
OLD فقط — NEW غير موجود في DELETE. OLD يحتوي الصف المحذوف.
الاتنين متاحين
٤. ما الفرق بين STABLE وIMMUTABLE وVOLATILE في PL/pgSQL Functions؟
كلهم نفس الشيء
VOLATILE (default): ممكن تغيّر DB أو تعطي نتائج مختلفة. STABLE: لا تغيّر DB، نفس النتيجة في نفس الـ Transaction. IMMUTABLE: دايماً نفس النتيجة لنفس المدخلات — الـ Planner يُحسّنها.
IMMUTABLE أبطأ دايماً
