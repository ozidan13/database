أوامر الـ DDL

CREATE • ALTER • DROP • TRUNCATE

CREATE TABLE
ALTER TABLE
DROP TABLE
TRUNCATE
CREATE TABLE بننشئ بيه جدول جديد. بنحدد اسم الجدول والأعمدة وأنواع البيانات والقيود.
PostgreSQL — CREATE TABLE
-- إنشاء جدول الأقسام CREATE TABLE department ( dept_id SERIAL PRIMARY KEY, dept_name VARCHAR(100) NOT NULL UNIQUE, location VARCHAR(100), budget DECIMAL(12,2) DEFAULT 0, created_at TIMESTAMP DEFAULT NOW() ); -- إنشاء جدول الموظفين مع FK CREATE TABLE employee ( emp_id SERIAL PRIMARY KEY, first_name VARCHAR(50) NOT NULL, last_name VARCHAR(50) NOT NULL, email VARCHAR(100) UNIQUE NOT NULL, salary DECIMAL(10,2) CHECK (salary > 0), hire_date DATE DEFAULT CURRENT_DATE, is_active BOOLEAN DEFAULT TRUE, dept_id INT REFERENCES department(dept_id) ON DELETE SET NULL );
ALTER TABLE بنعدل بيه جدول موجود — نضيف أعمدة، نحذف، نغير نوع البيانات.
PostgreSQL — ALTER TABLE
-- إضافة عمود جديد ALTER TABLE employee ADD COLUMN phone VARCHAR(20); -- تغيير نوع عمود ALTER TABLE employee ALTER COLUMN phone TYPE VARCHAR(30); -- إضافة قيد ALTER TABLE employee ADD CONSTRAINT chk_salary CHECK (salary BETWEEN 1000 AND 1000000); -- حذف عمود ALTER TABLE employee DROP COLUMN phone; -- إعادة تسمية جدول ALTER TABLE employee RENAME TO staff;
⚠️ DROP TABLE بيحذف الجدول بالكامل مع كل البيانات — مفيش رجعة!
PostgreSQL — DROP TABLE
-- حذف جدول DROP TABLE employee; -- حذف لو موجود (مش بيطلع error لو مش موجود) DROP TABLE IF EXISTS employee; -- حذف مع كل الـ Dependencies (FK وغيره) DROP TABLE department CASCADE; -- حذف Database كاملة DROP DATABASE mastery_db;
💡 دايماً استخدم IF EXISTS في الـ Scripts عشان متكسرش لو الجدول مش موجود.
TRUNCATE بيحذف كل البيانات من الجدول بس بيسيب الهيكل. أسرع من DELETE بكتير!
PostgreSQL — TRUNCATE
-- حذف كل البيانات وإعادة الـ SERIAL لـ 1 TRUNCATE TABLE employee RESTART IDENTITY; -- الفرق بين TRUNCATE و DELETE -- DELETE: بطيئة، قابلة للـ ROLLBACK، بتنشط الـ Triggers -- TRUNCATE: سريعة جداً، مش قابلة للـ ROLLBACK بسهولة

القيود — Constraints

Enforcing Data Integrity

PRIMARY KEY

يميز كل صف بشكل فريد. مش NULL + UNIQUE تلقائياً. كل جدول ليه PK واحد بس.

FOREIGN KEY

بيربط الجدول بجدول تاني. القيمة لازم تكون موجودة في الجدول الأصلي أو NULL.

NOT NULL

مينفعش الحقل يبقى فاضي. دايماً استخدمه على الحقول الأساسية زي name.

UNIQUE

كل القيم في العمود لازم تكون مختلفة. مثال: email — مينفعش يتكرر.

CHECK

شرط لازم يتحقق. مثال: salary > 0 أو age BETWEEN 18 AND 65.

DEFAULT

قيمة افتراضية لو مش حددنا حاجة. مثال: is_active DEFAULT TRUE.

⚙️ سلوك الـ Foreign Key عند الحذف

ON DELETE Options
-- CASCADE: احذف الصفوف المرتبطة تلقائياً dept_id INT REFERENCES department(dept_id) ON DELETE CASCADE -- SET NULL: بدّل الـ FK بـ NULL dept_id INT REFERENCES department(dept_id) ON DELETE SET NULL -- RESTRICT: منع الحذف لو في بيانات مرتبطة dept_id INT REFERENCES department(dept_id) ON DELETE RESTRICT -- NO ACTION: زي RESTRICT بس بيتأكد في آخر التعاملة dept_id INT REFERENCES department(dept_id) ON DELETE NO ACTION
