Subquery Examples
-- WHERE subquery: الموظفين اللي راتبهم أعلى من المتوسط SELECT first_name, salary FROM employee WHERE salary > (SELECT AVG(salary) FROM employee); -- IN subquery: موظفين في أقسام بميزانية أكتر من مليون SELECT first_name FROM employee WHERE dept_id IN ( SELECT dept_id FROM department WHERE budget > 1000000 ); -- FROM subquery: Derived Table SELECT dept_id, avg_sal FROM ( SELECT dept_id, AVG(salary) AS avg_sal FROM employee GROUP BY dept_id ) dept_stats WHERE avg_sal > 7000; -- Scalar subquery في الـ SELECT SELECT first_name, salary, (SELECT AVG(salary) FROM employee) AS company_avg, salary - (SELECT AVG(salary) FROM employee) AS diff_from_avg FROM employee; -- EXISTS: الموظفين اللي ليهم تبعيين (Dependents) SELECT first_name FROM employee e WHERE EXISTS ( SELECT 1 FROM dependent d WHERE d.emp_id = e.emp_id );

عمليات المجموعات

UNION • INTERSECT • EXCEPT

القاعدة الأساسية: الاستعلامين لازم يكون فيهم نفس عدد الأعمدة ونفس أنواع البيانات تقريباً!

UNION

دمج نتايج استعلامين — بيزيل التكرار تلقائياً. لو عايز التكرار استخدم UNION ALL.

INTERSECT

بس الصفوف المشتركة في الاستعلامين — التقاطع.

EXCEPT

الصفوف في الاستعلام الأول بس — بيحذف اللي موجود في التاني.

Set Operations
-- UNION: موظفين + مدراء (بدون تكرار) SELECT emp_id, first_name FROM employee UNION SELECT emp_id, first_name FROM manager; -- UNION ALL: بيحتفظ بالتكرار (أسرع) SELECT product_id FROM orders_2023 UNION ALL SELECT product_id FROM orders_2024; -- INTERSECT: الطلاب اللي حضروا كورس Python والـ SQL SELECT student_id FROM enrollment WHERE course_id = 1 INTERSECT SELECT student_id FROM enrollment WHERE course_id = 2; -- EXCEPT: الموظفين اللي مش مدراء SELECT emp_id FROM employee EXCEPT SELECT emp_id FROM manager;
