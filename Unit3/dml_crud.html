CRUD Operations

عمليات إضافة وقراءة وتعديل وحذف البيانات

الـ CRUD هو قلب كل تطبيق — كل حاجة بتتعمل هي في الآخر Insert أو Select أو Update أو Delete! 🎯

CRUD — الأربع عمليات الأساسية

Create Read Update Delete

C

CREATE → INSERT

إضافة بيانات جديدة للجدول. دايماً بنحدد الأعمدة والقيم.

R

READ → SELECT

قراءة وعرض البيانات. الأمر الأكتر استخداماً في حياتك كلها!

U

UPDATE → UPDATE

تعديل بيانات موجودة. تحذير: لو نسيت الـ WHERE — هتعدّل كل الجدول!

D

DELETE → DELETE

حذف بيانات. تحذير: تاني! لو نسيت الـ WHERE — هتحذف كل الجدول!

INSERT — إضافة بيانات

Adding New Records

INSERT Examples
-- إضافة صف واحد INSERT INTO employee (first_name, last_name, email, salary, dept_id) VALUES ('أحمد', 'محمد', 'ahmed@company.eg', 8500.00, 1); -- إضافة كذا صف مرة واحدة INSERT INTO employee (first_name, last_name, email, salary, dept_id) VALUES ('فاطمة', 'علي', 'fatma@company.eg', 9200.00, 1), ('عمر', 'حسن', 'omar@company.eg', 7800.00, 2), ('نور', 'سالم', 'nour@company.eg', 11000.00, 2); -- INSERT من SELECT (نسخ بيانات) INSERT INTO archive_employee SELECT * FROM employee WHERE is_active = FALSE;

SELECT — قراءة البيانات

Reading & Filtering Data

SELECT — From Basic to Advanced
-- كل الأعمدة SELECT * FROM employee; -- أعمدة محددة SELECT first_name, last_name, salary FROM employee; -- مع Alias (اسم مستعار) SELECT first_name || ' ' || last_name AS full_name, salary * 12 AS annual_salary FROM employee; -- مرتب تنازلياً بالراتب SELECT first_name, salary FROM employee ORDER BY salary DESC; -- أول 5 موظفين بأعلى راتب SELECT first_name, salary FROM employee ORDER BY salary DESC LIMIT 5; -- من row 11 لـ 20 (Pagination) SELECT * FROM employee ORDER BY emp_id LIMIT 10 OFFSET 10; -- بدون تكرار SELECT DISTINCT dept_id FROM employee;

WHERE — التصفية

Filtering with Conditions

=, !=, <>
مساواة / عدم مساواة
<, >, <=, >=
مقارنة الأرقام
BETWEEN a AND b
في نطاق معين
IN (v1, v2, ...)
في قائمة قيم
LIKE '%pattern%'
بحث بنمط — % أي عدد حروف
IS NULL / IS NOT NULL
فاضي / مش فاضي
AND, OR, NOT
دمج الشروط
EXISTS (subquery)
لو الاستعلام الفرعي بيرجع نتايج
WHERE Examples
-- راتب بين 5000 و 10000 SELECT * FROM employee WHERE salary BETWEEN 5000 AND 10000; -- اسمه بيبدأ بـ 'أحمد' SELECT * FROM employee WHERE first_name LIKE 'أحمد%'; -- في قسم 1 أو 3 SELECT * FROM employee WHERE dept_id IN (1, 3); -- شروط متعددة SELECT * FROM employee WHERE salary > 8000 AND is_active = TRUE AND dept_id = 2;
UPDATE & DELETE
-- تعديل راتب موظف معين UPDATE employee SET salary = 10000, is_active = TRUE WHERE emp_id = 5; -- زيادة 10% لكل موظفين قسم 2 UPDATE employee SET salary = salary * 1.10 WHERE dept_id = 2; -- حذف موظف معين DELETE FROM employee WHERE emp_id = 7; -- حذف الموظفين غير النشطين DELETE FROM employee WHERE is_active = FALSE AND hire_date < '2020-01-01';
⚠️ القاعدة الذهبية: قبل أي UPDATE أو DELETE — دايماً اعمل SELECT بنفس الـ WHERE الأول وتأكد من النتايج!

Quick Knowledge Check

❓ سؤال ١: إيه الأمر الصح لإضافة بيانات في جدول؟

ADD INTO employee VALUES (...);
INSERT INTO employee (col1, col2) VALUES (v1, v2);
CREATE employee VALUES (...);

❓ سؤال ٢: عايز أجيب أول 10 موظفين بأعلى راتب — إيه الأمر؟

SELECT * FROM employee ORDER BY salary DESC LIMIT 10;
SELECT TOP 10 * FROM employee ORDER BY salary;
SELECT * FROM employee LIMIT 10 ORDER BY salary DSC;

اللي فيه كلمة أحمد في النص
اللي بيبدأ بكلمة أحمد (% بعد = أي حاجة جاية)
اللي بينتهي بكلمة أحمد
