INSERT — Single, Multiple, and From SELECT

INSERT — Complete Reference
-- ══ INSERT بسيط — صف واحد ══ INSERT INTO department (dept_name, budget) VALUES ('Computer Science', 500000.00); -- ══ INSERT متعدد — صفوف كتير (أفضل للـ Performance) ══ INSERT INTO product (sku, product_name, price, stock_qty, category_id) VALUES ('LAPTOP-001', 'MacBook Pro 14', 59999.99, 50, 1), ('PHONE-001', 'iPhone 15 Pro', 44999.99, 100, 1), ('TABLET-001', 'iPad Pro', 29999.99, 75, 1); -- ده Bulk Insert — أسرع بكتير من 3 INSERTs منفصلين -- ══ RETURNING — اجيب البيانات بعد الإدراج ══ INSERT INTO customer (email, full_name) VALUES ('ahmed@example.com', 'أحمد محمود') RETURNING customer_id, created_at; -- بيرجع customer_id الجديد فوراً — مش محتاج SELECT تانية -- ══ INSERT ... SELECT — نقل بيانات ══ INSERT INTO archived_orders (order_id, customer_id, total, archived_at) SELECT order_id, customer_id, total_amount, NOW() FROM orders WHERE order_date < '2023-01-01' AND status = 'delivered'; -- ══ INSERT with DEFAULT ══ INSERT INTO orders (customer_id) -- order_date, status بيبقوا DEFAULT VALUES (42) RETURNING *; -- اجيب كل بيانات الأوردر الجديد
🏢 RETURNING في الـ Application Code

في Node.js/Python، بعد ما تعمل INSERT للـ Order، بتحتاج الـ order_id عشان تضيف ORDER_ITEMS. بدل ما تعمل INSERT ثم SELECT منفصلة، استخدم RETURNING:
INSERT INTO orders (...) VALUES (...) RETURNING order_id;
بترجع القيمة في نفس الـ query response — أسرع وأبسط.

⚠️ لا تحذف column names من الـ INSERT

❌ INSERT INTO product VALUES (1, 'test', 100)
لو حد بعدك أضاف عمود جديد للجدول — الـ INSERT ده هيفشل أو هيحط القيم في الأماكن الغلط!
✅ دايماً اكتب أسماء الأعمدة صراحةً.

🔍 SELECT — الاستعلام بالتفصيل

SELECT — Full Clause Reference with Execution Order

الـ SELECT هو أكثر أمر هتكتبه في حياتك. فيه ترتيب execution معين مختلف عن ترتيب الكتابة:

📋 ترتيب تنفيذ الـ SELECT (Logical Order)

1️⃣ FROM + JOINs → تحديد الجداول والدمج
2️⃣ WHERE → فلترة الصفوف (قبل التجميع)
3️⃣ GROUP BY → تجميع الصفوف
4️⃣ HAVING → فلترة المجموعات (بعد التجميع)
5️⃣ SELECT → اختيار الأعمدة (وتطبيق الدوال)
6️⃣ DISTINCT → إزالة التكرار
7️⃣ ORDER BY → ترتيب
8️⃣ LIMIT / OFFSET → تقليص النتيجة

SELECT — All Clauses Reference
-- ══ الشكل الكامل ══ SELECT [DISTINCT] columns_or_expressions FROM table_name [alias] [JOIN ...] [WHERE condition] [GROUP BY columns] [HAVING aggregate_condition] [ORDER BY columns [ASC|DESC]] [LIMIT n] [OFFSET m]; -- ══ أمثلة مع كل clause ══ -- 1. الكل مع ALIAS SELECT p.product_name AS "اسم المنتج", p.price AS "السعر", c.category_name AS "التصنيف" FROM product p JOIN category c ON p.category_id = c.category_id WHERE p.price > 1000 ORDER BY p.price DESC LIMIT 10; -- 2. DISTINCT SELECT DISTINCT specialty FROM doctor ORDER BY specialty; -- 3. WHERE بكل خياراته SELECT * FROM product WHERE price > 500 AND price <= 5000 -- range check AND category_id IN (1, 2, 3) -- IN list AND product_name LIKE '%Pro%' -- pattern matching AND stock_qty IS NOT NULL -- NULL check AND sku SIMILAR TO 'LAPTOP-%'; -- regex-like -- 4. BETWEEN (مكافئ >= AND <=) SELECT * FROM orders WHERE order_date BETWEEN '2024-01-01' AND '2024-03-31'; -- 5. ORDER BY متعدد الأعمدة SELECT full_name, salary, dept_id FROM employee ORDER BY dept_id ASC, salary DESC; -- القسم أولاً، الأعلى راتباً أولاً داخل القسم -- 6. LIMIT + OFFSET (Pagination) SELECT * FROM product ORDER BY product_id LIMIT 20 -- 20 منتج للصفحة OFFSET 40; -- ابدأ من الصف 41 (الصفحة 3) -- Formula: OFFSET = (page_num - 1) * page_size -- 7. Expressions في SELECT SELECT full_name, salary, salary * 12 AS annual_salary, ROUND(salary * 0.15, 2) AS tax, EXTRACT(YEAR FROM AGE(birth_date)) AS age, UPPER(full_name) AS name_upper, CURRENT_DATE - hire_date AS days_employed FROM employee; -- 8. CASE Expression SELECT product_name, price, CASE WHEN price < 1000 THEN 'اقتصادي' WHEN price < 5000 THEN 'متوسط' WHEN price < 20000 THEN 'راقي' ELSE 'فاخر' END AS price_category FROM product;
💡 LIKE vs ILIKE vs SIMILAR TO

LIKE '%ahmed%': Case-sensitive في PostgreSQL
ILIKE '%ahmed%': Case-INsensitive (PostgreSQL-specific)
SIMILAR TO '%ahmed|ali%': Regex-like
~ '^[A-Z]': Full Regex (أقوى — استخدمه بحذر، أبطأ على كميات كبيرة)

✏️ UPDATE — تحديث البيانات

UPDATE — Safe Patterns and Common Pitfalls

UPDATE — Complete Reference
-- ══ UPDATE بسيط ══ UPDATE product SET price = 49999.99 WHERE product_id = 1; -- ══ UPDATE متعدد الأعمدة ══ UPDATE employee SET salary = salary * 1.10, -- زيادة 10% updated_at = NOW() WHERE dept_id = 3 AND hire_date < '2022-01-01'; -- ══ UPDATE مع RETURNING ══ UPDATE orders SET status = 'shipped' WHERE order_id = 500 RETURNING order_id, status, customer_id; -- ══ UPDATE من جدول آخر ══ UPDATE product SET price = np.new_price FROM new_prices np WHERE product.sku = np.sku; -- ══ UPDATE بشرط subquery ══ UPDATE customer SET is_vip = TRUE WHERE customer_id IN ( SELECT customer_id FROM orders GROUP BY customer_id HAVING SUM(total_amount) > 100000 );
🚨 UPDATE بدون WHERE — أخطر أمر في قاعدة البيانات

❌ UPDATE employee SET salary = 0; → كل الرواتب بتبقى صفر!

✅ قبل أي UPDATE مهم:
Step 1: اعمل BEGIN; (Transaction)
Step 2: اعمل SELECT بنفس الـ WHERE عشان تشوف مين هيتأثر
Step 3: اعمل الـ UPDATE
Step 4: اتحقق من النتيجة
Step 5: إما COMMIT أو ROLLBACK

المحترفين دايماً بيشتغلوا في Transaction لأي تعديل مهم.

🗑️ DELETE والـ Soft Delete Pattern

DELETE vs Soft Delete — Real-World Data Deletion Strategy

DELETE — Patterns
-- ══ DELETE بسيط ══ DELETE FROM product WHERE product_id = 1; -- ══ DELETE مع RETURNING ══ DELETE FROM cart_item WHERE customer_id = 42 RETURNING *; -- شوف إيه اللي اتحذف -- ══ DELETE بشرط subquery ══ DELETE FROM orders WHERE status = 'cancelled' AND order_date < NOW() - INTERVAL '2 years'; -- ══ Soft Delete Pattern (بدل الحذف الحقيقي) ══ -- 1. أضف أعمدة للـ Soft Delete ALTER TABLE product ADD COLUMN is_deleted BOOLEAN NOT NULL DEFAULT FALSE, ADD COLUMN deleted_at TIMESTAMPTZ, ADD COLUMN deleted_by INT REFERENCES users(user_id); -- 2. Soft Delete فعلياً UPDATE product SET is_deleted = TRUE, deleted_at = NOW(), deleted_by = 7 -- id الموظف اللي حذف WHERE product_id = 99; -- 3. في كل الاستعلامات — استثني المحذوف SELECT * FROM product WHERE is_deleted = FALSE; -- أو IS NOT TRUE -- 4. View لإخفاء المحذوف تلقائياً CREATE VIEW active_products AS SELECT * FROM product WHERE is_deleted = FALSE;
🏢 Hard Delete vs Soft Delete — متى تستخدم كل منهم؟

Soft Delete (الأفضل في معظم الأنظمة):
✅ Audit Trail — تعرف مين حذف وإمتى
✅ GDPR Compliance — ممكن تسترجع بيانات
✅ منع كسر الـ Referential Integrity
✅ المستخدم قدر يغلط ويرجع

Hard Delete (متى يكون مناسب؟):
→ لوجز مؤقتة بتُحذف تلقائياً بعد فترة
→ GDPR "Right to be forgotten" — المستخدم طلب حذف بياناته فعلاً
→ بيانات ما ليهاش علاقات (كـ cache tables)

🔄 UPSERT — Insert or Update

ON CONFLICT — The Database Save Pattern

الـ Upsert بيعمل: "لو الصف موجود → حدّثه. لو مش موجود → أدرجه." مهم جداً في sync scenarios.

ON CONFLICT — UPSERT Patterns
-- ══ ON CONFLICT DO NOTHING ══ INSERT INTO student_phone (student_id, phone) VALUES (42, '+20123456789') ON CONFLICT (student_id, phone) DO NOTHING; -- لو الرقم موجود للطالب ده — تجاهل الطلب بدون error -- ══ ON CONFLICT DO UPDATE ══ INSERT INTO product (sku, product_name, price, stock_qty) VALUES ('LAPTOP-001', 'MacBook Pro 16', 64999.99, 30) ON CONFLICT (sku) DO UPDATE SET product_name = EXCLUDED.product_name, -- EXCLUDED = القيم الجديدة price = EXCLUDED.price, stock_qty = EXCLUDED.stock_qty, updated_at = NOW(); -- لو SKU موجود → حدّث. لو لأ → أضف. -- ══ Upsert مشروط ══ INSERT INTO inventory (product_id, warehouse_id, quantity) VALUES (1, 3, 100) ON CONFLICT (product_id, warehouse_id) DO UPDATE SET quantity = inventory.quantity + EXCLUDED.quantity -- لو موجود → أضف الكمية للموجودة. لو لأ → أضف جديد. WHERE inventory.quantity + EXCLUDED.quantity <= 10000; -- += فقط لو النتيجة مش هتتخطى 10000
⚡ Transactions — الـ ACID

Transactions — BEGIN, COMMIT, ROLLBACK, SAVEPOINT

الـ Transaction هي مجموعة عمليات بتُنفَّذ كوحدة واحدة — إما كلها تنجح أو كلها تفشل. ده أساس الـ ACID.

🔬 ACID — ما يعنيه كل حرف

A — Atomicity: إما كل العمليات بتنجح أو كلها بتُلغى.
C — Consistency: قاعدة البيانات دايماً في حالة صحيحة بعد كل transaction.
I — Isolation: الـ Transactions المتوازية مش بتشوف تعديلات بعضها (حسب isolation level).
D — Durability: بعد الـ COMMIT، التغييرات مش بتضيع حتى لو في crash.

Transactions — Bank Transfer Example
-- ══ مثال: تحويل بنكي — يجب أن يكون atomic ══ BEGIN; -- بدء الـ Transaction -- 1. خصم من حساب المرسِل UPDATE account SET balance = balance - 5000.00 WHERE account_id = 101 AND balance >= 5000.00; -- تحقق من الرصيد -- 2. تحقق من إن التحديث نجح (affectedRows) -- لو 0 rows → ROLLBACK -- 3. إضافة للمستقبِل UPDATE account SET balance = balance + 5000.00 WHERE account_id = 202; -- 4. تسجيل في سجل التحويلات INSERT INTO transfer_log (from_account, to_account, amount, transfer_dt) VALUES (101, 202, 5000.00, NOW()); COMMIT; -- كل حاجة نجحت → حفظ -- ══ لو في error في أي خطوة ══ ROLLBACK; -- إلغاء كل حاجة عملناها في الـ Transaction -- ══ SAVEPOINT — نقاط وسطية ══ BEGIN; INSERT INTO orders (customer_id) VALUES (1); SAVEPOINT after_order; -- نقطة حفظ مؤقتة INSERT INTO order_item (order_id, product_id, qty) VALUES (1,5,2); -- لو فشل ↓ ROLLBACK TO SAVEPOINT after_order; -- ارجع لنقطة بعد الأوردر -- الأوردر لسه موجود — فقط البند اتلغى COMMIT;
🏢 Transactions في الـ Application Code (Node.js / Python)

في أي Framework — بتبدأ Transaction في الـ Connection، وبتعمل BEGIN/COMMIT/ROLLBACK برمجياً. مثال في Node.js:
const client = await pool.connect();
await client.query('BEGIN');
try { ... await client.query('COMMIT'); }
catch { await client.query('ROLLBACK'); }

✍️ تمرين عملي

🛒 تمرين: منظومة إضافة أوردر في E-Commerce

عندنا جداول: customer, product, orders, order_item. اكتب الـ DML الكامل لـ:
1. عميل جديد اسمه "سارة أحمد" email سارة@مثال.com
2. أوردر جديد للعميل ده
3. بند في الأوردر: 2 قطعة من product_id = 5
4. لو الـ stock من المنتج ده أقل من 2 → عمل ROLLBACK
5. لو نجح كل حاجة → حدّث الـ stock وCOMMIT
E-Commerce Order Flow — Full Transaction
BEGIN; -- 1. إضافة العميل (أو تجاهله لو موجود) INSERT INTO customer (full_name, email) VALUES ('سارة أحمد', 'sara@example.com') ON CONFLICT (email) DO NOTHING RETURNING customer_id; -- خزّن الـ ID -- 2. تحقق من الستوك أولاً -- لو stock_qty < 2 → ROLLBACK في الـ Application SELECT stock_qty FROM product WHERE product_id = 5 AND stock_qty >= 2 FOR UPDATE; -- Lock الصف أثناء الـ Transaction -- 3. إنشاء الأوردر INSERT INTO orders (customer_id, status) VALUES ({customer_id_from_step_1}, 'confirmed') RETURNING order_id; -- 4. إضافة بند الأوردر INSERT INTO order_item (order_id, product_id, quantity, unit_price) SELECT {order_id_from_step_3}, 5, 2, price -- السعر وقت الشراء (مش بعدين) FROM product WHERE product_id = 5; -- 5. خصم من الستوك UPDATE product SET stock_qty = stock_qty - 2 WHERE product_id = 5; -- 6. تحديث total_amount في الأوردر UPDATE orders SET total_amount = ( SELECT SUM(quantity * unit_price) FROM order_item WHERE order_id = {order_id} ) WHERE order_id = {order_id}; COMMIT; -- كل حاجة نجحت -- لو حصل error في أي خطوة → ROLLBACK في Application
ليه FOR UPDATE مهم؟

FOR UPDATE بيعمل Row Lock على صف المنتج. يعني لو موظفين عاملين Transaction في نفس الوقت على نفس المنتج — التاني هيستنى لحد ما الأول يعمل COMMIT أو ROLLBACK. ده بيمنع Race Condition وبيضمن الـ Stock صح.

🧠 اختبار الفهم
١. الفرق بين WHERE وHAVING في ترتيب تنفيذ الـ SELECT؟
مفيش فرق — الاتنين بيعملوا نفس الحاجة
WHERE بتفلتر الصفوف قبل GROUP BY. HAVING بتفلتر المجموعات بعد GROUP BY.
HAVING أسرع من WHERE
٢. ليه RETURNING مفيد بعد INSERT؟
عشان نتحقق إن الـ INSERT نجح
نجيب الـ ID الجديد أو القيم المحسوبة (DEFAULT) من غير SELECT تانية
بيحسّن الـ Performance
٣. الـ Soft Delete pattern — ما الميزة الرئيسية على الحذف الفعلي؟
أسرع من DELETE
يحتفظ بالبيانات للـ Audit ويسمح بالاسترجاع ويحمي الـ Referential Integrity
يوفّر مساحة التخزين
٤. EXCLUDED في ON CONFLICT DO UPDATE يشير إلى؟
الصف الموجود في الجدول
القيم الجديدة اللي حاولنا نحطها في الـ INSERT المتعارض
