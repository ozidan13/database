🗄️ Schema Design — PostgreSQL Core

Relational Core: Users, Orders, Payments, Inventory — 3NF with Smart Denormalization

users

  • 🔑 id BIGSERIAL PK
  • email TEXT UNIQUE NN
  • password_hash TEXT
  • full_name TEXT
  • role TEXT (customer/admin)
  • created_at TIMESTAMPTZ
  • ▸ idx: email, role

orders

  • 🔑 id BIGSERIAL PK
  • → user_id FK users
  • status TEXT CHECK
  • total_amount NUMERIC(12,2)
  • shipping_address JSONB
  • order_date TIMESTAMPTZ
  • ▸ idx: user_id, status+date

order_items

  • 🔑 id BIGSERIAL PK
  • → order_id FK orders
  • product_id TEXT (Mongo ref)
  • product_name TEXT (snapshot)
  • quantity INT CHECK > 0
  • unit_price NUMERIC(10,2)
  • ▸ idx: order_id INCLUDE(price)

inventory

  • 🔑 product_id TEXT PK
  • quantity INT DEFAULT 0
  • reserved INT DEFAULT 0
  • warehouse_id INT
  • ▸ available = qty - reserved

payments

  • 🔑 id BIGSERIAL PK
  • → order_id FK orders
  • status TEXT CHECK
  • gateway TEXT
  • amount NUMERIC(12,2)
  • gateway_ref TEXT UNIQUE
  • ▸ idx: order_id, gateway_ref

reviews

  • 🔑 id BIGSERIAL PK
  • → user_id FK users
  • product_id TEXT
  • rating SMALLINT CHECK 1-5
  • body TEXT
  • UNIQUE(user_id, product_id)
  • ▸ idx: product_id, rating
PostgreSQL — Core Schema DDL
CREATE TABLE users ( id BIGSERIAL PRIMARY KEY, email TEXT UNIQUE NOT NULL, password_hash TEXT NOT NULL, full_name TEXT NOT NULL, role TEXT NOT NULL DEFAULT 'customer' CHECK (role IN ('customer','admin','seller')), created_at TIMESTAMPTZ NOT NULL DEFAULT NOW() ); CREATE TABLE orders ( id BIGSERIAL PRIMARY KEY, user_id BIGINT NOT NULL REFERENCES users(id), status TEXT NOT NULL DEFAULT 'pending' CHECK (status IN ('pending','confirmed','shipped','delivered','cancelled')), total_amount NUMERIC(12,2) NOT NULL CHECK (total_amount >= 0), shipping_address JSONB, order_date TIMESTAMPTZ NOT NULL DEFAULT NOW() ); CREATE TABLE inventory ( product_id TEXT PRIMARY KEY, -- MongoDB ObjectId as Text quantity INT NOT NULL DEFAULT 0 CHECK (quantity >= 0), reserved INT NOT NULL DEFAULT 0 CHECK (reserved >= 0), CONSTRAINT chk_reserved CHECK (reserved <= quantity) ); -- Indexes for common access patterns CREATE INDEX idx_orders_user_date ON orders(user_id, order_date DESC); CREATE INDEX idx_orders_status ON orders(status) WHERE status IN ('pending','confirmed','shipped'); CREATE INDEX idx_items_order ON order_items(order_id) INCLUDE (product_id, quantity, unit_price); CREATE INDEX idx_reviews_product ON reviews(product_id, rating);

placeOrder — PostgreSQL Transaction في Node.js
async function placeOrder( userId: number, items: Array<{ productId: string; quantity: number; price: number; name: string }> ) { const client = await pool.connect(); try { await client.query('BEGIN'); // 1. Lock and verify inventory for all items atomically for (const item of items) { const { rows } = await client.query( `SELECT quantity - reserved AS available FROM inventory WHERE product_id = $1 FOR UPDATE`, // Row-level lock [item.productId] ); if (!rows[0] || rows[0].available < item.quantity) throw new Error(`Insufficient stock: ${item.productId}`); } // 2. Create order + calculate total const total = items.reduce((s, i) => s + i.price * i.quantity, 0); const { rows: [order] } = await client.query( `INSERT INTO orders(user_id, total_amount) VALUES($1,$2) RETURNING id`, [userId, total] ); // 3. Insert order items (snapshot prices) await client.query( `INSERT INTO order_items(order_id,product_id,product_name,quantity,unit_price) SELECT $1, v.product_id, v.name, v.quantity, v.price FROM jsonb_to_recordset($2::jsonb) AS v(product_id text, name text, quantity int, price numeric)`, [order.id, JSON.stringify(items)] ); // 4. Reserve inventory for (const item of items) { await client.query( `UPDATE inventory SET reserved = reserved + $1 WHERE product_id = $2`, [item.quantity, item.productId] ); } // 5. Clear Redis cart await redis.del(`cart:${userId}`); await client.query('COMMIT'); return order.id; } catch (err) { await client.query('ROLLBACK'); throw err; } finally { client.release(); } }
🔍 Product Search — PostgreSQL Full-Text

Full-Text Search with tsvector, Faceted Filters, and Relevance Ranking

MongoDB Product + PostgreSQL FTS Sync
-- products_search: Sync table from MongoDB for Full-Text CREATE TABLE products_search ( mongo_id TEXT PRIMARY KEY, name TEXT NOT NULL, category TEXT, brand TEXT, price NUMERIC(10,2), avg_rating NUMERIC(3,2), search_vector tsvector GENERATED ALWAYS AS ( setweight(to_tsvector('arabic', coalesce(name,'')), 'A') || setweight(to_tsvector('arabic', coalesce(brand,'')), 'B') || setweight(to_tsvector('arabic', coalesce(category,'')), 'C') ) STORED ); CREATE INDEX idx_ps_fts ON products_search USING gin(search_vector); CREATE INDEX idx_ps_priceON products_search(price); -- قدرة البحث: نص + فلاتر + ترتيب بالصلة SELECT mongo_id, name, price, avg_rating, ts_rank(search_vector, query) AS relevance FROM products_search, plainto_tsquery('arabic', 'لابتوب ديل') query WHERE search_vector @@ query AND price BETWEEN 20000 AND 50000 AND avg_rating >= 4 ORDER BY relevance DESC, avg_rating DESC LIMIT 20;
📦 Inventory — Concurrency Safe

Dual-Column Inventory (quantity/reserved), Confirm Order, Cancel & Release

Inventory Lifecycle — Reserve → Confirm → Release
-- عند تأكيد الدفع: تحويل المحجوز لمكتسب UPDATE inventory SET quantity = quantity - $2, -- تقليل المخزون الفعلي reserved = reserved - $2 -- تحرير الحجز WHERE product_id = $1 AND quantity - reserved >= $2; -- Double check -- عند إلغاء الطلب: تحرير الحجز UPDATE inventory SET reserved = reserved - $2 WHERE product_id = $1; -- Cron Job: تحرير الحجوزات المنتهية (orders ماتأكدتش بعد 30 دقيقة) WITH expired AS ( UPDATE orders SET status = 'cancelled' WHERE status = 'pending' AND order_date < NOW() - INTERVAL '30 minutes' RETURNING id ) UPDATE inventory i SET reserved = i.reserved - oi.quantity FROM order_items oi JOIN expired e ON e.id = oi.order_id WHERE i.product_id = oi.product_id;
